interface IBabyGame{
    struct User {
        address inviter; //推荐人
        uint value; //有效
        uint scale; //等级
        uint lastSettleTime;
        bool reJoin; //是否自动入市, 也就是入金到期后自动续期

        uint inviteAwardValue; //直推奖
        uint scaleAwardValue;  //社区奖
        uint contributionAwardValue; //贡献奖

        uint inviteValue; //伞下总业绩
        address maxZone;  //大区
    }

    struct JoinItem {
        uint value; //本金
        uint createTime; //开始时间 
        uint perIssueTime; // 锁定时长
        uint dueTime; //结束时间, 当为自动续期时,为uint最大值
        uint rate;
    }

    struct Reparation {
        uint startTime;
        uint endTime;
        uint amount;
    }

    function expiredItemsValue() external view returns(uint amount, uint value);//amount: 用账户余额
    function getUser() external view returns(User memory); 
    function getJoinItems(uint skip, uint count) external view returns(JoinItem[] memory datas);//入金记录
    function getReparations() external view returns(Reparation[] memory);
    
    
    function join(uint amount, uint amoutOutMin, uint baseDays,  bool flag) external; //flag: true, 用账户余额,false: 用钱包余额 baseDays: 18
    function takeBack(address to) external; //第一页面里的的提现, 用账户余额 > 0时,提到钱包里

    function contributioRewardValue() external view returns(uint value); // 可提现的贡献奖
    function harvestContributioReward(address to) external; //我的社区里的提现
}

interface IBabyGame is IStructs{

    function expiredItemsValue(address user) external view returns(uint amount, uint value);//amount: 用账户余额
    function getUser(address user) external view returns(User memory); 
    function getJoinItems(address user, uint skip, uint count) external view returns(JoinItem[] memory datas);//入金记录
    function getReparations(address user) external view returns(Reparation[] memory);
    
    function isTopInviter(address user) external view returns(bool);

    function register(address inviter) external; //注册
    function reJoin(bool flag) external;   //设置自动做市
    function join(uint amount, uint amoutOutMin, uint baseDays,  bool flag) external; //flag: true, 用账户余额,false: 用钱包余额 baseDays: 18
    function takeBack(address to) external; //第一页面里的的提现, 用账户余额 > 0时,提到钱包里

    function contributioRewardValue(address user) external view returns(uint value); // 可提现的贡献奖
    function harvestContributioReward(address to) external; //我的社区里的提现
}
